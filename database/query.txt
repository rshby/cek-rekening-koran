query get data
SELECT
    DISTINCT A.TRDAT7, A.TRDAT6, A.TRTIME, A.TRDORC,
             CASE WHEN A.TELLERID IN ('DDCYCLING', 'SVCCHGPAY', 'TAXCHARGE') THEN ROUND(CONVERT(DECIMAL(20,7), A.amount), 2)
                  ELSE A.AMOUNT END AS AMOUNT,
             '0' BAL,
             A.TRREMK, A.TRREFN, A.ECINDICATOR, A.TRREFNEC, A.TELLERID, B.ATMID, B.ATMJOURNALSEQ, A.JOURNALSEQ, A.SOURCEOFBRANCH
FROM
    ABCS_T_ABCSHIST A LEFT JOIN ABCS_T_ATMTRN_BAK B ON A.TRREFN = B.TRREFN
WHERE
        A.accountnumber= '045202000114806'
ORDER BY A.TRDAT7 ASC, A.TRTIME ASC
